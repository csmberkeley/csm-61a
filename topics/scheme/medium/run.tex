	)

\begin{solution}
\begin{blocksection}
\begin{lstlisting}
(define duplicate
	(lambda (lst n)
		(cond ((null? lst) '())
			((< n (car lst)) 
				(cons (car lst) (duplicate lst (+ n 1)))
			)
          		(else 
          			(duplicate (cdr lst) 0)
	   		)
     		)
   	)
)
\end{lstlisting}

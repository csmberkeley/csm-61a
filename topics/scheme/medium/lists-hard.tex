\question What will Scheme output? If it outputs a list, draw a box-and-pointer diagram. If a part contains multiple expressions, only
give the output of the final expression. Assume each expression is evaluated sequentially.

\begin{parts}
\begin{blocksection}
\part
\begin{lstlisting}
scm> (cons 3 4)
\end{lstlisting}

\begin{solution}[.5in]
\begin{lstlisting}
(3 . 4)
\end{lstlisting}
\end{solution}

\part
\begin{lstlisting}
scm> (cons 3 (cons (cons 4 5) nil))
\end{lstlisting}

\begin{solution}[.5in]
\lstinline$(3 (4 . 5))$
\end{solution}

\part 
\begin{lstlisting}
scm> (3 . 4)
\end{lstlisting}

\begin{solution}[.5in]
\lstinline$Error$
\end{solution}

\part 
\begin{lstlisting}
scm> '(3 . 4)
\end{lstlisting}

\begin{solution}[.5in]
\lstinline$(3 . 4)$
\end{solution}

\part
\begin{lstlisting}
scm> (list 3 4 5)
\end{lstlisting}

\begin{solution}[.5in]
\lstinline$(3 4 5)$
\end{solution}

\part 
\begin{lstlisting}
scm> (list 3 '(4 5) 6)
\end{lstlisting}

\begin{solution}[.5in]
\lstinline$(3 (4 5) 6)$
\end{solution}

\end{blocksection}

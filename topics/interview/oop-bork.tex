class Dog:
    rating = 11
    def __init__(self, name)
        self.name = name
    def speak(self):
        print("woof!")
    def praise(self):
        self.rating = self.rating + 1
		
class Pom(Dog):
    def __init__(self, speech):
        self.speech = speech
    def speak(self):
        print(self.speech)

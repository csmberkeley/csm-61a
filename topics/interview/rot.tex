\begin{blocksection}
\begin{solution}
\begin{lstlisting}
def rotate(t):
  branch_labels = [b.label for b in t.branches]
  n = len(t.branches)
  for i in range(n):
    branch = t.branches[i]
    branch.label = branch_labels[(i + 1) % n]
    rotate(branch)
\end{lstlisting}
\end{solution}

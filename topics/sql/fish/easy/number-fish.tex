\begin{blocksection}
\part Write a query to find the total number of fish in the ocean. Additionally, include the number of species we summed. Your output should have the number of species and the total population and it should yield in a table with a single row.

\begin{solution}[0.7in]
\begin{lstlisting}
SELECT COUNT(species), SUM(pop) 
FROM fish;
\end{lstlisting}
\end{solution}
\end{blocksection}

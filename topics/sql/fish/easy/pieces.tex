\begin{blocksection}
\part Write a query which is going to yield a table which has a row for each species, the species's price/piece in our hatchery, and then the species's price/piece in our competitors' hatchery.

\begin{solution}
    \begin{lstlisting}
        SELECT fish.species, fish.price, competitor.price
        FROM fish, competitor
        WHERE fish.species = competitor.species;
    \end{lstlisting}
\end{solution}
\end{blocksection}

\begin{solution}
\begin{lstlisting}
def root_subtraction(t):
    total = t.label
    for b in t.branches:
        if not b.is_leaf():
            total -= b.label
            total -= (b.label - root_subtraction(b))
    return total
\end{lstlisting}

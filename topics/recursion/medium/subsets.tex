\begin{blocksection}
\begin{solution}
\begin{lstlisting}
    if n == 0:
        return [[]]
    if len(lst) == n:
        return [lst]
    with_first = [[lst[0]] + x for x in subsets(lst[1:], n - 1)]
    without_first = subsets(lst[1:], n)
    return with_first + without_first
\end{lstlisting}
\end{solution}

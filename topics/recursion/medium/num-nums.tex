    """

def num_nums(d, x, t):
    if d == 0 and t > 0:
        return 0
    elif d == 0 and t == 0:
        return 1
    elif t < 0:
        return 0
    else:
        with_d_here = num_nums(d - 1, x, t - 1)
        without_d_here = 8 * num_nums(d - 1, x, t)
        return with_d_here + without_d_here
\end{lstlisting}

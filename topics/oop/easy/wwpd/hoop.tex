class Baller:
    all_players = []
    def __init__(self, name, has_ball = False):
        self.name = name
        self.has_ball = has_ball
        Baller.all_players.append(self)

    def pass_ball(self, other):
        if self.has_ball:
            self.has_ball = False
            other.has_ball = True
            return True
        else:
            return False

class BallHog(Baller):
    def pass_ball(self, other):
        return False

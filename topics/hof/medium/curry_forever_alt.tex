    10

\begin{solution}
\begin{lstlisting}
def curry_forever (f, arg_num, base=0):
    def curr(x):
        return curry_forever(f, arg_num - 1, f(x, base))
    if arg_num == 0:
        return base
    else:
        return curr

\end{lstlisting}

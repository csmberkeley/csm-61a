def linkify_rec(lst):
    if not lst:
        return Link.empty
    else:
        return Link(lst[0], linkify_rec(lst[1:]))

def linkify_iter(lst):
    retVal = Link.empty
    for elem in lst[::-1]:
        retVal = Link(elem, retVal)
    return retVal

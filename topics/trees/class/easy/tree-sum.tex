\begin{blocksection}
\question Implement \lstinline$tree_sum$ which takes in a Tree
object and replaces the root value with the sum of all the values
in the tree. \lstinline$tree_sum$ should also return the new root
value.

\begin{lstlisting}
def tree_sum(t):
    """
    >>> t = Tree(1, [Tree(2, [Tree(3)]), Tree(4)])
    >>> tree_sum(t)
    10
    >>> t.root
    10
    >>> t.branches[0].root
    5
    >>> t.branches[1].root
    4
    """
\end{lstlisting}

\begin{solution}[1in]
\begin{lstlisting}
    for b in t.branches:
        t.root += tree_sum(b)
    return t.root
\end{lstlisting}
\end{solution}
\end{blocksection}

    \begin{solution}[0.7in]
    \begin{lstlisting}
def is_clone(t):
    if t.is_leaf():
        return True
    if len(t.branches) != 2:
        return False
    left, right = t.branches
    if not (is_clone(left) and is_clone(right)):
        return False
    def count_leaves(t):
        if t.is_leaf():
            return 1
        return sum(count_leaves(b) for b in t.branches)
    return count_leaves(left) == count_leaves(right)
    \end{lstlisting}

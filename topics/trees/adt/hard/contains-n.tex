\begin{solution}
\begin{lstlisting}
    if n == 0:
        return True
    elif n == 1 and is_leaf(t) and label(t) == elem:
        return True
    elif is_leaf(t):
        return False
    elif label(t) == elem:
        return True in [contains_n(elem, n - 1, b) for b in     
          branches(t)]
    else:
        return True in [contains_n(elem, n, b) for b in 
          branches(t)]

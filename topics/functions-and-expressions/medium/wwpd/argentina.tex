\begin{lstlisting}
def square(x): 
    return x * x 
		
def argentina(n):
    print(n) 
    if n > 0: 
        return lambda k: k(n+1) 
    else: 
        return 1 / n
\end{lstlisting}

After executing the above code, what will Python
display when evaluating each of the following expressions?

\begin{lstlisting}
>>> print(1, print(2))
\end{lstlisting}

\begin{lstlisting}
2
1 None
\end{lstlisting}
\end{solution}

\end{lstlisting}

\end{lstlisting}
\end{solution}

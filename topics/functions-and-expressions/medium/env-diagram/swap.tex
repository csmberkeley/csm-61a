\question Draw the environment diagram that results from running the following code.

\begin{lstlisting}
def swap(x, y):
    x, y = y, x
    return print("Swapped!", x, y)

x, y = 60, 1
a = swap(x, y)
swap(a, y)
\end{lstlisting}

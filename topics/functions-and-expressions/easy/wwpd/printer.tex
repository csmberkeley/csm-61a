\question 
Consider the following function definition:
\begin{lstlisting}
def printer():
	x = 3
	count = 0
	while count < 3:
		x = print(x)
		count += 1
\end{lstlisting}

What will the Python interpreter display when
evaluating \lstinline{printer()}?

\begin{solution}[0.5in]
\begin{verbatim}
3
None
None
\end{verbatim}
\end{solution}
\end{blocksection}

\begin{blocksection}
\question Write the number of calls to \lstinline$scheme_eval$ and
\lstinline$scheme_apply$ for the code below.

% \begin{parts}
\begin{lstlisting}
(if 1 (+ 2 3) (/ 1 0))
\end{lstlisting}

\begin{tabular}{lrrrr}
\lstinline$scheme_eval$ & 1 & 3 & 4 & 6 \\
\lstinline$scheme_apply$ & 1 & 2 & 3 & 4
\end{tabular}
\\

\begin{solution}
6 \lstinline$scheme_eval$, 1 \lstinline$scheme_apply$.
\end{solution}

\vspace{2\baselineskip}
\begin{lstlisting}
(or #f (and (+ 1 2) 'apple) (- 5 2))
\end{lstlisting}

\begin{tabular}{lrrrr}
\lstinline$scheme_eval$ & 6 & 8 & 9 & 10 \\
\lstinline$scheme_apply$ & 1 & 2 & 3 & 4
\end{tabular}
\\

\begin{solution}
8 \lstinline$scheme_eval$, 1 \lstinline$scheme_apply$.
\end{solution}

\vspace{2\baselineskip}
\begin{lstlisting}
(define (square x) (* x x))

(+ (square 3) (- 3 2))
\end{lstlisting}

\begin{tabular}{lrrrr}
\lstinline$scheme_eval$ & 2 & 5 & 14 & 24 \\
\lstinline$scheme_apply$ & 1 & 2 & 3 & 4
\end{tabular}

\begin{solution}
14 \lstinline$scheme_eval$, 4 \lstinline$scheme_apply$.
\end{solution}

\vspace{2\baselineskip}
\begin{lstlisting}
(define (add x y) (+ x y))

(add (- 5 3) (or 0 2))
\end{lstlisting}

\begin{solution}[1in]
13 \lstinline$scheme_eval$, 3 \lstinline$scheme_apply$.
\end{solution}

\begin{blocksection}
\question Write the number of calls to \lstinline$scheme_eval$ and
\lstinline$scheme_apply$ for the code below.

% \begin{parts}
 \begin{lstlisting}
scm> (+ 1 2)
3
\end{lstlisting}

\begin{solution}[1in]
4 \lstinline$scheme_eval$, 1 \lstinline$scheme_apply$.
\end{solution}

 \begin{lstlisting}
scm> (if 1 (+ 2 3) (/ 1 0))
5
\end{lstlisting}

\begin{solution}[1in]
6 \lstinline$scheme_eval$, 1 \lstinline$scheme_apply$.
\end{solution}

 \begin{lstlisting}
scm> (or #f (and (+ 1 2) 'apple) (- 5 2))
apple
\end{lstlisting}

\begin{solution}[1in]
8 \lstinline$scheme_eval$, 1 \lstinline$scheme_apply$.
\end{solution}

\begin{lstlisting}
scm> (define (square x) (* x x))
square
scm> (+ (square 3) (- 3 2))
10
\end{lstlisting}

\begin{solution}[1in]
14 \lstinline$scheme_eval$, 4 \lstinline$scheme_apply$.
\end{solution}

 \begin{lstlisting}
scm> (define (add x y) (+ x y))
add
scm> (add (- 5 3) (or 0 2))
2
\end{lstlisting}

\begin{solution}[1in]
13 \lstinline$scheme_eval$, 3 \lstinline$scheme_apply$.
\end{solution}
% \end{parts}
\end{blocksection}

\begin{blocksection}
\begin{solution}[0in]

\begin{lstlisting}
def eras_cycle(songs, k):
    n = len(songs)
    for i in range(k):  # Loop for k items
        yield songs[i % n]  # Restart when reaching the end
\end{lstlisting}

\end{solution}

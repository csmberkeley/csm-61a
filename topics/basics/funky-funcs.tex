\question For the following expressions, simplify the operands in the order of evaluation of the entire expression

Example: \lstinline{add(3, mul(4, 5))}

Order of Evaluation: \lstinline{add(3, mul(4, 5))} $\rightarrow$ \lstinline{add(3, 20)} $\rightarrow$ \lstinline{23}

\begin{parts}
\part \begin{lstlisting}
add(1, mul(2, 3))
\end{lstlisting}
\begin{solution}[.2in]
    \lstinline{add(1, mul(2, 3))} $\rightarrow$ \lstinline{add(1, 6)} $\rightarrow$ \lstinline{7}
\end{solution}

\part \begin{lstlisting}
add(mul(2, 3), add(1, 4))
\end{lstlisting}
\begin{solution}[.2in]
    \lstinline{add(mul(2, 3), add(1, 4))} $\rightarrow$ \lstinline{add(6, add(1, 4))} $\rightarrow$ \lstinline{add(6, 5)} $\rightarrow$ \lstinline{11}
\end{solution}

\part \begin{lstlisting}
max(mul(1, 2), add(5, 6), 3, mul(mul(3, 4), 1), 7)
\end{lstlisting}
\begin{solution}[.2in]
    \lstinline{max(mul(1, 2), add(5, 6), 3, mul(mul(3, 4), 1), 7)} $\rightarrow$ 
    \lstinline{max(2, add(5, 6), 3, mul(mul(3, 4), 1), 7)} $\rightarrow$
    \lstinline{max(2, 11, 3, mul(mul(3, 4), 1), 7)} $\rightarrow$
    \lstinline{max(2, 11, 3, mul(12, 1), 7)} $\rightarrow$
    \lstinline{max(2, 11, 3, 12, 7)} $\rightarrow$
    \lstinline{max(2, 11, 3, 12, 7)} $\rightarrow$ \lstinline{12}
\end{solution}
\end{parts}

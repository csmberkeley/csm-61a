\begin{blocksection}
\begin{solution}
\begin{lstlisting}
def min_fact(n, margin):
    product, n = n, n - 1
    while product <= margin and n > 0:
        product, n = product * n, n - 1
    if product <= margin:
        return -1
    return product
\end{lstlisting}
\end{solution}

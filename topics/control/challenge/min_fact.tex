\begin{blocksection}
\begin{solution}
\begin{lstlisting}
def fact_limit(n, limit):
    if n > limit:
        return -1
    product = n
    n = n - 1
    while product * n <= limit and n > 0:
        product, n = product * n, n - 1
    return product
\end{lstlisting}
\end{solution}

\begin{blocksection}
\question Write a function that returns \lstinline{True} if a number is divisible by 4 and \lstinline{False} otherwise.




% \vspace{2in}



\begin{solution}[1in]
\begin{lstlisting}
def is_divisible_by_4(num):
    return num % 4 == 0
\end{lstlisting}
\end{solution}

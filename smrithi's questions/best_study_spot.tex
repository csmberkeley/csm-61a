\begin{solution}[0.7in]
\begin{lstlisting}
def best_study_spot(t, key):
    if t.is_leaf():
        return t.label
    best = t.label
    for b in t.branches:
        candidate = best_study_spot(b, key)
        if key(candidate) > key(best):
            best = candidate
    return best
\end{lstlisting}
\end{solution}
